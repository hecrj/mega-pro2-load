ALGORITMO DE SELECCIÓN DE RUTA DE DESCARGA

Ruta encontrar_ruta(Peticion &p, Red &r, int suma_bandas, int prof)
{
	if(s.esta_libre() and s.tiene_pelicula(p.consultar_pelicula()))
	{
		int total_velocidad = s.consultar_velocidad() + suma_bandas;

		if(p.consultar_peso() / total_velocidad <= 1)
		{
			Ruta ruta(r.raiz(), prof);
			ruta.dar_prioridad();
		}
		else
		{
			if(r.tiene_hijos())
			{
				Ruta r1 = encontrar_ruta(p, r.obtener_hijo_izq(), total_velocidad, prof+1);

				if(r.unico_hijo() or r1.tiene_prioridad())
					return r1;

				Ruta r2 = encontrar_ruta(p, r.obtener_hijo_der(), total_velocidad, prof+1);

				if(r2.tiene_prioridad())
					return r2;

				if(r1.consultar_velocidad() == r2.consultar_velocidad())
				{
					if(r1.consultar_profundidad() == r2.consultar_profundidad())
				}
			}
			else
				Ruta ruta(r.raiz(), prof);
		}
	}

	return ruta;
}
